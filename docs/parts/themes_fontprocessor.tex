		%%Add entries into the registry
		%%extracts the current values assigned to the variables
		\prop_put:Nen{\table_names}{#2}{\verb|#1| & \texttt{\tl_to_str:c{l__#2_family}}& \texttt{\tl_to_str:c{l__#2_style}}}

	}
